\begin{frame}[fragile,t]
Four different payment and receipt timing combinations
\begin{enumerate}
	\item \textcolor{magenta}{Outright purchase}: ordinary transaction
	\item \textcolor{magenta}{Fully leveraged purchase}: investor borrows the full amount
	\item \textcolor{magenta}{Prepaid forward contract}: pay today, receive the share later
	\item \textcolor{magenta}{Forward contract}: agree on price now, pay/receive later
\end{enumerate}
