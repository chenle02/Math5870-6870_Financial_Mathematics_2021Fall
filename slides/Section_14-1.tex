%-------------- start slide -------------------------------%{{{ 1
\begin{frame}[fragile,t]
	\begin{center}
		An \textcolor{magenta}{exotic option},  or \textcolor{magenta}{nonstandard option}, is simply an
		option with some contractual difference from standard options.

		\bigskip

		Exotic options solve particular business problems that\\
		an ordinary option do not.

		\bigskip
		They are often constructed by tweaking ordinary options in minor ways.
	\end{center}
\end{frame}
%-------------- end slide -------------------------------%}}}
%-------------- start slide -------------------------------%{{{ 1
\begin{frame}[fragile,t]
	\frametitle{Questions}
	\begin{itemize}
		\item How does the exotic payoff compare to that of a standard option?
			\bigskip
		\item Can the exotic option be approximated by a portfolio of other options?
			\bigskip
		\item Is the exotic option cheap or expensive relative to standard options?
			\bigskip
		\item What is the rationale for the use of the exotic option?
			\bigskip
		\item How easily can the exotic option be hedged?
	\end{itemize}
\end{frame}
